Submit Checkboxes

     data-ctat-tutor="false"

<div id="checkB" class="CTATCheckBox"
     data-ctat-tutor="false"

     data-ctat-tutor="false"
     name="submitGroup">no</div>

     data-ctat-target="submitGroup">Submit</div>
